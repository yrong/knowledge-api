create table discussions(
	id serial not null,
    idcode uuid not null,
    created_at timestamp without time zone,
    fromer text not null,
    toer text[],
    content text, 
   CONSTRAINT discussions_pkey PRIMARY KEY (id)
);
COMMENT ON TABLE public.discussions
    IS '文章评论';
COMMENT ON COLUMN public.discussions.id
    IS '自增主键';
COMMENT ON COLUMN public.discussions.idcode
    IS '文章idcode';
COMMENT ON COLUMN public.discussions.created_at
    IS '评论时间';
COMMENT ON COLUMN public.discussions.fromer
    IS '评论者';
COMMENT ON COLUMN public.discussions.toer
    IS '被@者';
COMMENT ON COLUMN public.discussions.content
    IS '评论内容';
    
    
create index idx_discussions_idcode on discussions using btree(idcode);
